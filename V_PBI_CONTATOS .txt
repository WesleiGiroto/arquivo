CREATE OR REPLACE VIEW V_PBI_CLIENTES AS
SELECT
  C.CLIENTE,
    C.NOME AS NOME_CLIENTE,
	CC.CONTATO AS COD_CONTATO,
    C.DDD_TELEFONE,
    C.TELEFONE AS NUM_TELEFONE,
    C.DDD_CELULAR,
    C.CELULAR AS NUM_CELULAR,

    (C.DDD_TELEFONE || ' - ' || C.TELEFONE) AS TELEFONE,
    (C.DDD_CELULAR || ' - ' || C.CELULAR) AS CELULAR,

    C.TIPOVIA_ENTREGA,
    C.LOGRADOURO_ENTREGA,
    (C.TIPOVIA_ENTREGA || ': ' || C.LOGRADOURO_ENTREGA) AS ENDEREÇO,

    C.BAIRRO_ENTREGA AS BAIRRO,
    C.MUNICIPIO_ENTREGA,
    C.UF_ENTREGA,
    (C.MUNICIPIO_ENTREGA || ' - ' || C.UF_ENTREGA) AS CIDADE,

    C.DTA_CADASTRO,
    CC.DES_CONTATO,


    F.EMPRESA,
    F.REVENDA,

    (TO_CHAR(F.EMPRESA) || ' - ' || TO_CHAR(F.REVENDA)) AS CHAVE_EMP_REV

FROM CNP.FAT_CLIENTE C

LEFT JOIN V_PBI_FATURAMENTO_VEICULOS F
       ON F.CLIENTE = C.CLIENTE

LEFT JOIN CNP.CAC_CONTATO CC
       ON CC.CLIENTE = C.CLIENTE

WHERE 1=1
    -- OTIMIZAÇÃO: FILTRO DOS ÚLTIMOS 3 ANOS
    AND C.DTA_CADASTRO >= ADD_MONTHS(TRUNC(SYSDATE), -36);
