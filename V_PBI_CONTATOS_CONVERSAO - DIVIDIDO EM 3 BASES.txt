
--------------------------------------- 1 BASE — SOMENTE CONTATOS ------------------
  
  CREATE OR REPLACE VIEW V_PBI_CONTATO_CONTATOS AS
  
  -- TABELA DIVIDIDA DA V_PBI_CONTATOS_CONVERSAO
  
SELECT
    C.EMPRESA,
    C.REVENDA,
    C.CHAVE_EMP_REV_CONTATO,
    C.CHAVE_EMP_REV,
    C.CHAVE_EMP_REV_USUARIO,
    C.CONTATO,
    C.USUARIO_ENCAMINHADO,
    C.DTA_CONTATO,
    C.DEPARTAMENTO_CONT,
    C.SITUACAO_CONTATO,
    C.DES_CONTATO,

    C.VENDEDOR,
    C.FORMA_CONTATO,
    C.DES_FORMA_CONTATO,

    
    CC.USUARIO_ENCAMINHADO_ORIGINAL AS VENDEDOR_ORIGINAL_REF

FROM V_PBI_CONTATOS C

INNER JOIN CNP.CAC_CONTATO CC
       ON CC.EMPRESA  = C.EMPRESA
      AND CC.REVENDA  = C.REVENDA
      AND CC.CONTATO  = C.CONTATO

WHERE EXTRACT(YEAR FROM C.DTA_CONTATO) >= (EXTRACT(YEAR FROM SYSDATE) - 5);



------------------------------------//// 2 CONTATOS + FATURAMENTO ///////---------------------------------------

 

CREATE OR REPLACE VIEW V_PBI_CONTATOS_FATURAMENTO AS

-- TABELA DIVIDIDA DA V_PBI_CONTATOS_CONVERSAO

SELECT
     C."EMPRESA",C."REVENDA",C."CHAVE_EMP_REV",C."CHAVE_EMP_REV_CONTATO",C."CHAVE_EMP_REV_USUARIO",C."VENDEDOR",C."USUARIO_ENCAMINHADO",C."CONTATO",C."DTA_CONTATO",C."DEPARTAMENTO_CONT",C."SITUACAO_CONTATO",C."DES_CONTATO",C."FORMA_CONTATO",C."DES_FORMA_CONTATO",

    FV.CHAVE_VEICULO,
    FV.DTA_ENTRADA_SAIDA,
    FV.DTA_DOCUMENTO,
    FV.MODALIDADE,
    FV.FATOPERACAO,
    FV.OPERACAO,
    FV.CAIXA,
    FV.CLIENTE,
	FV.NOME AS NOME_CLIENTE,
    FV.TELEFONE AS TELEFONE_CLIENTE,
    FV.CLIENTE_FATURAMENTO,
    FV.DEPARTAMENTO_FAT,
    FV.USUARIO,
    FV.TOT_MERCADORIA,
    FV.TOT_CUSTO_MEDIO,
    FV.TOT_QUANTIDADES,
    FV.VEICULO_FATURAMENTO,
    FV.VAL_TOTAL,
    FV.VAL_CUSTO,
    FV.VAL_OPCIONAIS,
    FV.VAL_DESCONTO,
    FV.VAL_FRETE,
    FV.VAL_DEVERIA
	
FROM V_PBI_CONTATOS C

INNER JOIN V_PBI_FATURAMENTO_VEICULOS FV
    ON FV.CHAVE_EMP_REV_CONTATO = C.CHAVE_EMP_REV_CONTATO
INNER JOIN V_PBI_VEICULOS V
    ON V.CHAVE_VEICULO = FV.CHAVE_VEICULO

INNER JOIN CNP.FAT_CLIENTE FC
    ON FV.CLIENTE = FV.CLIENTE
	   
WHERE EXTRACT(YEAR FROM FV.DTA_ENTRADA_SAIDA) >= (EXTRACT(YEAR FROM SYSDATE) - 5);
