
------------------------------------ VIEW COM FILTRO PARA RETIRAR DUPLICADAS -----------------------


CREATE OR REPLACE VIEW V_PBI_FATURAMENTO_VEICULOS AS

SELECT *
FROM (
    SELECT
        -- Chaves e Identificadores
        (FMV.EMPRESA || ' - ' || FMV.VEICULO) AS CHAVE_VEICULO,
        (FMC.EMPRESA || ' - ' || FMC.REVENDA) AS CHAVE_EMP_REV,
        (FMC.EMPRESA || ' - ' || FMC.REVENDA || ' - ' || FMC.CONTATO ) AS CHAVE_EMP_REV_CONTATO,
        FMC.EMPRESA,
        FMC.REVENDA,
        T2.NOME_EMPRESA,

        -- Dados da Transação
        FMC.DTA_ENTRADA_SAIDA,
        FMC.DTA_DOCUMENTO,
        FMC.MODALIDADE,
        FMC.TIPO_TRANSACAO,
        FMC.FATOPERACAO,
        FMC.OPERACAO,
        FMC.CAIXA,

        -- Clientes e Vendedores
        FMC.CLIENTE,
        FC.NOME AS NOME_CLIENTE,
        FC.TELEFONE AS TELEFONE_CLIENTE,
        FMC.CLIENTE_FATURAMENTO,
        FMC.DEPARTAMENTO AS DEPARTAMENTO_FAT,
        FMC.USUARIO,
        VEND.CHAVE_EMP_REV_VENDEDOR,
        VEND.CHAVE_EMP_REV_USUARIO,
        FMC.CONTATO,

        -- Métricas e Valores
        FMC.TOT_MERCADORIA,
        FMC.TOT_CUSTO_MEDIO,
        FMC.TOT_QUANTIDADES,
        FMV.VEICULO AS VEICULO_FATURAMENTO,
        FMV.VAL_TOTAL,
        FMV.VAL_CUSTO,
        FMV.VAL_OPCIONAIS,
        FMV.VAL_DESCONTO,
        FMV.VAL_FRETE,
        FMV.VAL_DEVERIA,

        -- CONTROLE DE DUPLICIDADE
        ROW_NUMBER() OVER (
            PARTITION BY FMC.USUARIO
            ORDER BY FMC.DTA_ENTRADA_SAIDA DESC
        ) AS RN

    FROM CNP.FAT_MOVIMENTO_CAPA FMC

    INNER JOIN CNP.FAT_MOVIMENTO_VEICULO FMV
        ON FMV.EMPRESA = FMC.EMPRESA
        AND FMV.REVENDA = FMC.REVENDA
        AND FMV.NUMERO_NOTA_FISCAL = FMC.NUMERO_NOTA_FISCAL
        AND FMV.SERIE_NOTA_FISCAL = FMC.SERIE_NOTA_FISCAL
        AND FMV.TIPO_TRANSACAO = FMC.TIPO_TRANSACAO
        AND FMV.CONTADOR = FMC.CONTADOR

    INNER JOIN CNP.VEI_VEICULO VEI
        ON VEI.EMPRESA = FMV.EMPRESA
        AND VEI.VEICULO = FMV.VEICULO

    INNER JOIN CNP.GER_USUARIO USU
        ON USU.USUARIO = FMC.USUARIO

    INNER JOIN CNP.FAT_TIPO_TRANSACAO FTT
        ON FTT.TIPO_TRANSACAO = FMC.TIPO_TRANSACAO

    LEFT JOIN CNP.FAT_CLIENTE FC
        ON FC.CLIENTE = FMC.CLIENTE

    INNER JOIN V_PBI_REVENDAS T2
        ON FMC.EMPRESA = T2.EMPRESA
        AND FMC.REVENDA = T2.REVENDA

    INNER JOIN V_PBI_VENDEDORES VEND
        ON VEND.USUARIO = USU.USUARIO
        AND VEND.EMPRESA = FMC.EMPRESA
        AND VEND.REVENDA = FMC.REVENDA

    WHERE FMC.TIPO_TRANSACAO IN ('V21','V07','U21','U07')
      AND EXTRACT(YEAR FROM FMC.DTA_ENTRADA_SAIDA) >= (EXTRACT(YEAR FROM SYSDATE) - 3)
      AND FMC.STATUS = 'F'
)

-- AQUI REMOVE OS DUPLICADOS:
WHERE RN = 1;
