CREATE OR REPLACE VIEW V_PBI_CONTATOS_CONVERSAO AS
SELECT
    C.EMPRESA,
    C.REVENDA,
    C.CHAVE_EMP_REV_CONTATO,
    C.CHAVE_EMP_REV,
    C.CHAVE_EMP_REV_USUARIO,
    C.CONTATO,
    C.USUARIO_ENCAMINHADO,
    C.DTA_CONTATO,
    C.DEPARTAMENTO_CONT,
    C.SITUACAO_CONTATO,
    C.DES_CONTATO,
    C.VENDEDOR,
    C.FORMA_CONTATO,
    C.DES_FORMA_CONTATO,
    FV.CHAVE_VEICULO,
    FV.DTA_ENTRADA_SAIDA,
    FV.DTA_DOCUMENTO,
    FV.MODALIDADE,
    FV.FATOPERACAO,
    FV.OPERACAO,
    FV.CAIXA,
    FV.CLIENTE,
    FV.CLIENTE_FATURAMENTO,
    FV.DEPARTAMENTO_FAT,
    FV.USUARIO,
    FV.TOT_MERCADORIA,
    FV.TOT_CUSTO_MEDIO,
    FV.TOT_QUANTIDADES,
    FV.VEICULO_FATURAMENTO,
    FV.VAL_TOTAL,
    FV.VAL_CUSTO,
    FV.VAL_OPCIONAIS,
    FV.VAL_DESCONTO,
    FV.VAL_FRETE,
    FV.VAL_DEVERIA,
    V.CHAVE_CHASSI,
    V.VEICULO,
    V.COR,
    V.CHASSI,
    V.MODELO,
    V.DES_MODELO,
    V.SITUACAO,
    V.NOVO_USADO,
    V.VAL_COMPRA,
    V.CATEGORIA_VEICULO,
    V.DES_CATEGORIA_VEICULO
FROM V_PBI_CONTATOS C
LEFT JOIN V_PBI_FATURAMENTO_VEICULOS FV ON FV.CHAVE_EMP_REV_CONTATO = C.CHAVE_EMP_REV_CONTATO
LEFT JOIN V_PBI_VEICULOS V ON V.CHAVE_VEICULO = FV.CHAVE_VEICULO

WHERE 1=1
				-- novo filtro 
		AND C.DTA_CONTATO >= ADD_MONTHS(TRUNC(SYSDATE), -36)
		
    -- OTIMIZAÇÃO: FILTRO DOS ÚLTIMOS 5 ANOS NA DATA DO CONTATO
    AND EXTRACT(YEAR FROM C.DTA_CONTATO) >= (EXTRACT(YEAR FROM SYSDATE) - 5);
